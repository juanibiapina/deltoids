<section class="landing-section" id="pager">
  <div class="landing-container">
    <h2 class="landing-section-title">Use it anywhere</h2>
    <p class="landing-section-lead">
      <code>deltoids</code> reads a unified diff on stdin and writes it back
      with each hunk expanded. Pipe <code>git diff</code>,{' '}
      <code>gh pr diff</code>, or set it as the pager for git or{' '}
      <a href="https://github.com/jesseduffield/lazygit">lazygit</a>.
    </p>
    <Tabs>
      <TabItem label="Pipe">
        <Code code="git diff | deltoids | less -R" lang="bash" />
      </TabItem>
      <TabItem label="GitHub PR">
        <Code code="gh pr diff <number> | deltoids | less -R" lang="bash" />
      </TabItem>
      <TabItem label="Git pager">
        <Code code="git config --global core.pager 'deltoids | less -R'" lang="bash" />
      </TabItem>
      <TabItem label="Lazygit">
        <Code code={`# ~/.config/lazygit/config.yml
git:
  paging:
    pager: deltoids`} lang="yaml" />
      </TabItem>
    </Tabs>
    <figure class="landing-figure">
      <Picture
        src={lazygitImg}
        formats={['avif', 'webp']}
        widths={[800, 1600, 2400]}
        sizes="(max-width: 90rem) calc(100vw - 2rem), 88rem"
        alt="Lazygit with deltoids configured as the pager, showing a diff in the right pane."
      />
      <figcaption><a href="https://github.com/jesseduffield/lazygit">Lazygit</a> with <code>deltoids</code> as its pager.</figcaption>
    </figure>
    <details class="landing-faq">
      <summary>How is this different from <code>git diff -W</code>?</summary>
      <p>
        <code>git diff -W</code> finds scope with regex (the
        <code>xfuncname</code> patterns from gitattributes).{' '}
        <code>deltoids</code> parses the file with tree-sitter.
      </p>
    </details>
